串行的异步和并行的异步

参考链接#

  • Promise.all()
  • Promise.allSettled()

串行的异步#

常规方式 Array.prototype.reduce() + promise.then()#

// 假想有一个 promise 生成函数function takePromise (number) {  // 这里使用 takePromise 函数创建一个 closure(闭包)，返回一个函数，此函数执行后会创建一个 promise 函数。  // 设计意图是，可以在当前 promise 执行前，通过 ...lastParams 接收上个 promise 执行成功后传过来的参数。  return (...lastParams) => new Promise(function (resolve) {    setTimeout(function () {      console.log(number, ...lastParams)      // 1 undefined      // 2 1      // 3 2      resolve(number)    }, 500)  })}
let promise = [takePromise(1), takePromise(2), takePromise(3)].reduce((previousPromise, currentPromiseFn) => {  return previousPromise.then((...lastParams) => {    return currentPromiseFn(...lastParams)  })}, Promise.resolve())
console.time('async')promise.then(function (result) {  console.log(result) // 3  console.timeEnd('async') // async: 1510.7578125 ms})

分析代码中的 promise chain#

  1. 在第 1 次遍历时，previousPromise 是 Promise.resolve()，currentPromiseFn 是 takePromise(1)。 可以理解为 Promise.resolve().then((...lastParams) => takePromise(1)(...lastParams))。
  2. 在第 2 次遍历时，previousPromise 是上次的 promise 链，即 Promise.resolve().then((...lastParams) => takePromise(1)(...lastParams))，currentPromiseFn 是 takePromise(2)。 可以理解为 Promise.resolve().then((...lastParams) => takePromise(1)(...lastParams)).then((...lastParams) => takePromise(2)(...lastParams))。
  3. 在第 2 次遍历时，previousPromise 是上次的 promise 链，即 Promise.resolve().then((...lastParams) => takePromise(1)(...lastParams)).then((...lastParams) => takePromise(2)(...lastParams))，currentPromiseFn 是 takePromise(3)。 可以理解为 Promise.resolve().then((...lastParams) => takePromise(1)(...lastParams)).then((...lastParams) => takePromise(2)(...lastParams)).then((...lastParams) => takePromise(3)(...lastParams))。

这种方式应用了 Promise 链式调用的原理：在 promise 链中的 then 回调函数中 return 另一个 promise，会造成 promise 链中断，产生新的 promise 链。

简洁方式 Array.prototype.reduce() + async/await#

function takePromise (number) {    return (...lastParams) => new Promise(function (resolve) {      setTimeout(function () {        console.log(number, ...lastParams)        resolve(number)      }, 500)    })  }
  let promise = [takePromise(1), takePromise(2), takePromise(3)].reduce(async (previousPromise, currentPromiseFn) => {    const lastParams = await previousPromise    return currentPromiseFn(lastParams)  }, Promise.resolve())
  console.time('async')  promise.then(function (result) {    console.log(result) // 3    console.timeEnd('async') // async: 1510.7578125 ms  })

并行的异步#

Promise.all() 写法#

const promise1 = Promise.resolve(1)const promise2 = 2const promise3 = new Promise((resolve, reject) => {  setTimeout(resolve, 1000, 3)})
console.time('async')Promise.all([promise1, promise2, promise3]).then((result) => {  console.log(result) // [1, 2, 3]  console.timeEnd('async') // async: 1011.548095703125 ms})

Promise.all() 存在一个问题#

如果这些 promise 中出现一个 reject，Promise.all 最终的返回的是错误消息

const promise1 = Promise.resolve(1)const promise2 = 2const promise3 = new Promise((resolve, reject) => {  setTimeout(reject, 1000, 3) // 这里我们换成 reject})
console.time('async')Promise.all([promise1, promise2, promise3]).then((result) => {  console.log(result) // 没执行}).catch((error) => {  console.error(error) // 3  console.timeEnd('async') // async: 1004.153076171875 ms})

Promise.allSettled() 写法#

const promise1 = Promise.resolve(1)const promise2 = 2const promise3 = new Promise((resolve, reject) => {  setTimeout(reject, 1000, 3) // 这里我们换成 reject})
console.time('async')Promise.allSettled([promise1, promise2, promise3]).then((result) => {  console.log(result)  // [  //   {"status":"fulfilled","value":1},  //   {"status":"fulfilled","value":2},  //   {"status":"rejected","reason":3}  // ]  console.timeEnd('async') // async: 1006.906005859375 ms})
