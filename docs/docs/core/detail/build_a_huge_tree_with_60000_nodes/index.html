构建一个有六万个节点的五层树结构

需求#

一个大数据量的树，深度是 5 层，大概有 2,0000-10,0000 个节点，以 可展开表格 的形式展示。

方案#

五层遍历#

通过嵌套五层 for 循环函数，创建 9**5=59049 个节点的树

export const dataFn = () => {  const threshold = 9;  const defaultNode = () => ({children: [], name: null})  const defaultLatestNode = () => ({name: null})
  let nodeTop = defaultNode()  for (let i = 1; i <= threshold; i++) {    let node1 = defaultNode()    let name1 = [i].join('-')    node1.key = name1    node1.name = name1    nodeTop.children.push(node1)    for (let j = 1; j <= threshold; j++) {      let node2 = defaultNode()      let name2 = [name1, j].join('-')      node2.key = name2      node2.name = name2      node1.children.push(node2)      for (let k = 1; k <= threshold; k++) {        let node3 = defaultNode()        let name3 = [name2, k].join('-')        node3.key = name3        node3.name = name3        node2.children.push(node3)        for (let l = 1; l <= threshold; l++) {          let node4 = defaultNode()          let name4 = [name3, l].join('-')          node4.key = name4          node4.name = name4          node3.children.push(node4)          for (let m = 1; m <= threshold; m++) {            let node5 = defaultLatestNode()            let name5 = [name4, m].join('-')            node5.key = name5            node5.name = name5            node4.children.push(node5)          }        }      }    }  }  return nodeTop.children}console.log(dataFn())
