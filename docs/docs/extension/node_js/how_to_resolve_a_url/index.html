Query string#

  • Query string - querystring.parse & querystring.stringify

URL#

解析为对象#

const url = require('url')url.parse('https://user:pass@sub.example.com:8080/p/a/t/h?query=string#hash')
{  "protocol": "https:",  "slashes": true,  "auth": "user:pass",  "host": "sub.example.com:8080",  "port": "8080",  "hostname": "sub.example.com",  "hash": "#hash",  "search": "?query=string",  "query": "query=string",  "pathname": "/p/a/t/h",  "path": "/p/a/t/h?query=string",  "href": "https://user:pass@sub.example.com:8080/p/a/t/h?query=string#hash"}

格式化为字符串#

const url = require('url')url.format({  protocol: 'https:',  slashes: true,  auth: 'user:pass',  host: 'sub.example.com:8080',  port: '8080',  hostname: 'sub.example.com',  hash: '#hash',  search: '?query=string',  query: 'query=string',  pathname: '/p/a/t/h',  path: '/p/a/t/h?query=string',  href: 'https://user:pass@sub.example.com:8080/p/a/t/h?query=string#hash',})
'https://user:pass@sub.example.com:8080/p/a/t/h?query=string#hash'

Query string#

解析为对象#

const querystring = require('querystring')querystring.parse('foo=bar&abc=xyz&abc=123')
{  "foo": "bar",  "abc": [    "xyz",    "123"  ]}

格式化为字符串#

const querystring = require('querystring')querystring.stringify({  foo: 'bar',  abc: [ 'xyz', '123' ]})
'foo=bar&abc=xyz&abc=123'

URL 编码#

编码#

encodeURIComponent('//test.baidu.com') // '%2F%2Faccount.baidu.com'

解码#

decodeURIComponent('%2F%2Faccount.baidu.com') // '//account.baidu.com'
