使用 Vue 3 Composition API 实现一个修改密码弹框

参考链接#

  • Composition API - setup
  • component props
  • useForm 基本表单
  • @ant-design-vue/use

分析和设计#

假想场景#

假想我们需要实现两个修改密码的弹框：

  1. 在首次登录时，重置初始密码。
  2. 在用户登录后，修改密码。

为了区别这两个弹框，我们将弹框分别命名为 重置初始密码弹框 和 修改密码弹框。

分析场景#

两个弹框的共同点是：

  1. 都有旧的密码、新的密码、确认密码三个输入框。
  2. 验证逻辑相同（三个密码框都必填、新的密码和确认密码要求输入一致）。

两个弹框的差异点是：

  1. 重置初始密码弹框。要求强制修改密码，不允许关闭弹框。
  2. 修改密码弹框。不允许关闭弹框。

简述设计#

我们把组件划分为 调用位置 和 组件实例位置。

  1. isInitialPassword 控制是否重置初始密码。我们在 调用位置 的 setup 中通过 setTimeout 中设置 isInitialPassword.value=true 实现模拟异步获取参数的功能。
  2. prop 传入参数。我们在 调用位置 通过 props 控制：1. 标题 title。 2. 是否可以点击遮罩层关闭弹框 maskClosable。 3. 是否显示关闭按钮 closable。
  3. ref 获取组件实例。我们在 调用位置 的 mounted 回调中通过 ref 获取弹框组件实例。
  4. state 控制显隐。我们在 调用位置，通过 ref 拿到组件实例，然后修改组件内部 visible 状态，来控制弹框显隐。
  5. validate 相关的逻辑被封装在 组件实例位置。根据 关注点分离 原则，组件实例位置 关注验证逻辑，调用位置 关注数据。
  6. event 接收数据。我们在 组件实例位置 验证数据通过后，使用 commit 事件来接收 组件实例位置 的数据 oldPassword, newPassword。

实现代码#

调用位置的代码#

<template>  <ChangePasswordDialog    ref="refResetPassword"    title="重置初始密码"    :maskClosable="false"    :closable="false"    @commit="resetPassword"  />  <ChangePasswordDialog    ref="refChangePassword"    title="修改密码"    @commit="changePassword"  /></template><script>import { defineComponent, ref, watch } from 'vue'import ChangePasswordDialog from './change-password-modal/index.vue'
export default defineComponent({  components: {    ChangePasswordDialog,  },  setup() {    // Composition API 很方便，可以在代码层面，把密切关联的逻辑放置在接近的位置        // 以下是密切关联的异步控制弹框显隐逻辑，放置在一起    const isInitialPassword = ref(false)    watch(() => isInitialPassword.value, (value) => { // 我们      if (value) refResetPassword.value.visible = true    })    new Promise(function(resolve) { // 这里是测试代码，在实际项目中替换为异步请求接口的逻辑      setTimeout(function () {        isInitialPassword.value = true        resolve()      }, 1000)    })
    // 以下是密切关联的事件接收数据逻辑，放置在一起    const refResetPassword = ref(null)    const refChangePassword = ref(null)    const resetPassword = ({oldPassword, newPassword}) => {      console.log(oldPassword, newPassword)    }    const changePassword = ({oldPassword, newPassword}) => {      console.log(oldPassword, newPassword)    }
    return {      refResetPassword,      refChangePassword,      resetPassword,      changePassword,    }  }})</script>

组件实例位置的代码#

<template>  <a-modal    v-model:visible="visible"    :title="title"    :maskClosable="maskClosable"    :closable="closable"    :confirm-loading="loading"  >    <template #footer>      <a-button type="primary" @click="commit">确定</a-button>      <a-button @click="resetFields">重置</a-button>    </template>    <a-form      ref="refForm"      :model="formState"      :label-col="labelCol"      :wrapper-col="wrapperCol"    >      <a-form-item v-bind="validateInfos.oldPassword" required has-feedback label="旧的密码">        <a-input autocomplete="off" type="password" v-model:value="formState.oldPassword" />      </a-form-item>      <a-form-item v-bind="validateInfos.newPassword" required has-feedback label="新的密码">        <a-input autocomplete="off" type="password" v-model:value="formState.newPassword" />      </a-form-item>      <a-form-item v-bind="validateInfos.confirmPassword" required has-feedback label="确认密码">        <a-input autocomplete="off" type="password" v-model:value="formState.confirmPassword" />      </a-form-item>    </a-form>  </a-modal></template>
<script>import { defineComponent, reactive, ref, toRaw } from 'vue'import { useForm } from '@ant-design-vue/use'
export default defineComponent({  name: 'ChangePasswordDialog',  props: {    title: { default: '' },    maskClosable: { default: true },    closable: { default: true },  },  emits: ['commit'],  setup(props, { emit }) {    // 以下是密切关联的弹框逻辑，放置在一起    const visible = ref(false)    const loading = ref(false)
    // 以下是密切关联的表单逻辑，放置在一起    const refForm = ref();    const formState = reactive({      oldPassword: '',      newPassword: '',      confirmPassword: '',    });    let validatePass = async (rule, value) => {      if (value === '') return Promise.reject('请输入新的密码');      if (formState.confirmPassword !== '') refForm.value.validateField('confirmPassword');      return Promise.resolve();    };    let validatePass2 = async (rule, value) => {      if (value === '') return Promise.reject('请再次输入新的密码');      if (value !== formState.newPassword) return Promise.reject("两次输入的密码不一致！");      return Promise.resolve();    };    const rules = reactive({      oldPassword: [{ required: true, trigger: 'change', message: '请输入旧的密码' }],      newPassword: [{ validator: validatePass, trigger: 'change' }],      confirmPassword: [{ validator: validatePass2, trigger: 'change' }],    });    const { resetFields, validate, validateInfos } = useForm(formState, rules);    const commit = async () => {      try {        await validate()        emit('commit', toRaw(formState))      } catch (error) {        console.log('error', error);      }    }
    return {      // modal      loading,      visible,
      // form      refForm,      rules,      formState,      labelCol: { span: 5 },      wrapperCol: { span: 16 },      validateInfos,      resetFields,      commit,    }  }})</script>
