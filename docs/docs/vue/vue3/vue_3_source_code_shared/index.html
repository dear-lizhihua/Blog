Vue 3 源码 shared 模块

重要声明#

为了易读，直接用 JavaScript 重写了。

一些模块#

@vue/shared - toTypeString#

export const objectToString = Object.prototype.toStringexport const toTypeString = (value) => objectToString.call(value)

用途#

进行类型检测

Object.prototype.toString.call(new Map()) // "[object Map]"Object.prototype.toString.call(new Set()) // "[object Set]"

@vue/shared - is#

export const isArray = Array.isArrayexport const isMap = (val) => toTypeString(val) === '[object Map]' // 依赖上面的 toTypeString 函数export const isSet = (val) => toTypeString(val) === '[object Set]' // 依赖上面的 toTypeString 函数
export const isDate = (val) => val instanceof Dateexport const isFunction = (val) => typeof val === 'function'export const isString = (val) => typeof val === 'string'export const isSymbol = (val) => typeof val === 'symbol'export const isObject = (val) => val !== null && typeof val === 'object'
export const isPromise = (val) => {  return isObject(val) && isFunction(val.then) && isFunction(val.catch)}
export const isPlainObject = (val) => toTypeString(val) === '[object Object]'
export const isIntegerKey = (key) =>  isString(key) &&  key !== 'NaN' &&  key[0] !== '-' &&  '' + parseInt(key, 10) === key

@vue/shared - makeMap#

export function makeMap(str, expectsLowerCase) {  const map = Object.create(null)  const list = str.split(',')  for (let i = 0; i < list.length; i++) {    map[list[i]] = true  }  return expectsLowerCase ? val => !!map[val.toLowerCase()] : val => !!map[val]}

用途#

用于创建一个 map，判断是否有某个字段。

const isABC = makeMap('a,b,c')console.log(isABC('a')) // trueconsole.log(isABC('b')) // trueconsole.log(isABC('c')) // trueconsole.log(isABC('d')) // false
